# === Importing Required Libraries ===
import requests
from bs4 import BeautifulSoup
import pandas as pd
import sqlite3
import numpy as np
from datetime import datetime

# === 1. Extract Function ===
def extract(url, table_attribs):
    """
    Extracts GDP data from the specified URL using BeautifulSoup.
    Returns a DataFrame with raw GDP values in millions.
    """
    response = requests.get(url)
    soup = BeautifulSoup(response.text, 'html.parser')
    table = soup.find('table', attrs=table_attribs)

    rows = table.find_all('tr')[1:]  # Skip header row
    data = []
    for row in rows:
        cols = row.find_all(['td', 'th'])
        if len(cols) >= 2:
            country = cols[0].text.strip()
            gdp_text = cols[1].text.strip().replace(',', '').replace('$', '')
            try:
                gdp = float(gdp_text)
                data.append([country, gdp])
            except ValueError:
                continue  # Skip rows with non-numeric GDP
    df = pd.DataFrame(data, columns=['Country', 'GDP_USD_million'])
    return df

# === 2. Transform Function ===
def transform(df):
    """
    Converts GDP from millions to billions.
    Rounds to 2 decimal places and drops the original column.
    """
    df['GDP_USD_billion'] = np.round(df['GDP_USD_million'] / 1000, 2)
    df.drop(columns=['GDP_USD_million'], inplace=True)
    return df

# === 3. Load to CSV Function ===
def load_to_csv(df, csv_path):
    """
    Saves the transformed DataFrame to a CSV file.
    """
    df.to_csv(csv_path, index=False)

# === 4. Load to Database Function ===
def load_to_db(df, conn, table_name):
    """
    Loads the DataFrame into a SQLite database table.
    """
    df.to_sql(table_name, conn, if_exists='replace', index=False)

# === 5. Run Query Function ===
def run_query(query_statement, conn):
    """
    Executes a SQL query and prints the results.
    """
    cursor = conn.cursor()
    cursor.execute(query_statement)
    results = cursor.fetchall()
    for row in results:
        print(row)

# === 6. Logging Function ===
def log_process(message):
    """
    Logs messages with timestamps to a log file.
    """
    timestamp = datetime.now().strftime('%Y-%m-%d %H:%M:%S')
    with open('etl_project_log.txt', 'a') as f:
        f.write(f'[{timestamp}] {message}\n')

# === Execution Flow ===
#if _name_ == "_main_":

#declaring variables
    url = 'https://web.archive.org/web/20230902185326/https://en.wikipedia.org/wiki/List_of_countries_by_GDP_%28nominal%29'
    table_attribs = {'class': 'wikitable'}
    csv_path = 'Countries_by_GDP.csv'
    db_path = 'World_Economies.db'
    table_name = 'Countries_by_GDP'
    query_statement = 'SELECT * FROM Countries_by_GDP WHERE GDP_USD_billion > 100'

    log_process('ETL process started.')
    df_extracted = extract(url, table_attribs)
    log_process('Data extraction completed.')

    df_transformed = transform(df_extracted)
    log_process('Data transformation completed.')

    load_to_csv(df_transformed, csv_path)
    log_process('Data loaded to CSV.')

    conn = sqlite3.connect(db_path)
    load_to_db(df_transformed, conn, table_name)
    log_process('Data loaded to database.')

    log_process('Running query for economies > 100 billion USD.')
    run_query(query_statement, conn)
    log_process('Query execution completed.')

    conn.close()
    log_process('ETL process finished.')
