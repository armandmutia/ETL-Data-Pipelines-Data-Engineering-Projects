##REFACTORED VERSION

import sqlite3
import pandas as pd

# ---------- Utility Functions ----------

def load_csv_to_db(conn, file_path, table_name, attribute_list, replace=True):
    """Load a CSV file into a SQLite table."""
    df = pd.read_csv(file_path, names=attribute_list)
    df.to_sql(table_name, conn, if_exists='replace' if replace else 'append', index=False)
    print(f"Table '{table_name}' is ready")

def run_query(conn, query):
    """Execute a SQL query and print results."""
    result = pd.read_sql(query, conn)
    print(query)
    print(result)
    return result

def append_data(conn, table_name, data_dict):
    """Append a dictionary of data into a table."""
    df = pd.DataFrame(data_dict)
    df.to_sql(table_name, conn, if_exists='append', index=False)
    print(f"Data appended successfully to '{table_name}'")

# ---------- Main Script ----------

# Connect to the SQLite3 service
conn = sqlite3.connect('STAFF.db')

# --- INSTRUCTOR Table ---
instructor_table = 'INSTRUCTOR'
instructor_attrs = ['ID', 'FNAME', 'LNAME', 'CITY', 'CCODE']
instructor_csv = '/home/project/INSTRUCTOR.csv'

load_csv_to_db(conn, instructor_csv, instructor_table, instructor_attrs)

# Queries
run_query(conn, f"SELECT * FROM {instructor_table}")
run_query(conn, f"SELECT FNAME FROM {instructor_table}")
run_query(conn, f"SELECT COUNT(*) FROM {instructor_table}")

# Append new instructor
new_instructor = {
    'ID': [100],
    'FNAME': ['John'],
    'LNAME': ['Doe'],
    'CITY': ['Paris'],
    'CCODE': ['FR']
}
append_data(conn, instructor_table, new_instructor)

# Verify row count
run_query(conn, f"SELECT COUNT(*) FROM {instructor_table}")

# --- Departments Table ---
dept_table = 'Departments'
dept_attrs = ['DEPT_ID', 'DEP_NAME', 'MANAGER_ID', 'LOC_ID']
dept_csv = '/home/project/Departments.csv'

load_csv_to_db(conn, dept_csv, dept_table, dept_attrs)

# Queries
run_query(conn, f"SELECT * FROM {dept_table}")
run_query(conn, f"SELECT DEP_NAME FROM {dept_table}")
run_query(conn, f"SELECT COUNT(*) FROM {dept_table}")

# Append new department
new_dept = {
    'DEPT_ID': [9],
    'DEP_NAME': ['Quality Assurance'],
    'MANAGER_ID': ['30010'],
    'LOC_ID': ['L0010']
}
append_data(conn, dept_table, new_dept)

# Verify row count
run_query(conn, f"SELECT COUNT(*) FROM {dept_table}")

# Close connection
conn.close()
